namespace = lpr_events

# ============================================================================
# CLIMATE EVENTS
# ============================================================================
# These events represent global "climate" changes that occur when worldwide
# trends (Independence, Innovation, Uprising) reach critical thresholds of 100.
# They affect all countries and represent major global shifts in political climate.
# ============================================================================

# Global Independence Climate - triggers when worldwide independence reaches 100
lpr_events.33 = {
    type = country_event
    placement = root

    title = lpr_events.33.t
    desc = lpr_events.33.d
    flavor = lpr_events.33.f

    event_image = {
        video = "unspecific_world_fair"
    }

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
    icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

    duration = 5

    trigger = {
        check_global_variable = {
            name = actionables_lpr_independence
            value >= 100
        }
    }

    # Option A: Embrace the independence wave
    option = {
        name = lpr_events.33.a
        default_option = yes
        add_modifier = { name = lpr_global_independence_climate
            months = 12
        }
        change_variable = {
            name = actionables_lpr_liberty
            add = 2
        }
    }

    # Option B: Resist the global trend
    option = {
        name = lpr_events.33.b
        add_modifier = { name = lpr_resistance_penalty
            months = 6
        }
        change_variable = {
            name = actionables_lpr_revolution
            add = 1
        }
    }
}

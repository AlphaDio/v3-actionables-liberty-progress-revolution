namespace = lpr_events

# ============================================================================
# FINISHER EVENTS
# ============================================================================
# These events trigger when a country accumulates 12 points in any LPR category
# (Liberty, Progress, or Revolution). With monthly builder events (3 points each),
# this typically happens after 4 monthly builder events where the player
# consistently chooses the same path (taking about 4 months).
# 
# The finisher events provide significant rewards but also reduce the accumulated
# points, creating a cycle of building up and cashing out.
# ============================================================================

# Finisher event - triggers when any LPR variable reaches 12 points
# This represents a yearly assessment where accumulated choices manifest
lpr_events.5 = {
    type = country_event
    placement = root

    title = lpr_events.5.t
    desc = lpr_events.5.d
    flavor = lpr_events.5.f

    event_image = {
        video = "unspecific_politicians_arguing"
    }

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
    icon = "gfx/interface/icons/event_icons/event_scales.dds"

    duration = 5

    cooldown = {
        years = 3
    }

    # Option A: Financial Stability Path
    # Rewards: Moderate treasury boost (15,000 gold)
    # Costs: Reduces Liberty points (local) and Independence (global) by 10
    # This path favors economic stability over political change
    option = {
        name = lpr_events.5.a
        default_option = yes
        trigger = {
            has_variable = actionables_lpr_liberty
            var:actionables_lpr_liberty >= 10
        }
        lpr_finisher_liberty_effect = yes
    }

    # Option B: National Unity Path
    # Rewards: Converts large radicals to loyalists (improves stability)
    # Costs: Reduces Progress points (local) and Innovation (global) by 10
    # This path focuses on social cohesion and reducing radicalism
    option = {
        name = lpr_events.5.b
        trigger = {
            has_variable = actionables_lpr_progress
            var:actionables_lpr_progress >= 10
        }
        lpr_finisher_progress_effect = yes
    }

    # Option C: Population Expansion Path
    # Rewards: Creates 50,000 new pops in a random state (matching capital culture/religion)
    # Costs: Reduces Revolution points (local) and Uprising (global) by 10
    # This path increases population, which can boost economic output
    option = {
        name = lpr_events.5.c
        trigger = {
            has_variable = actionables_lpr_revolution
            var:actionables_lpr_revolution >= 10
        }
        lpr_finisher_revolution_effect = yes
    }
}

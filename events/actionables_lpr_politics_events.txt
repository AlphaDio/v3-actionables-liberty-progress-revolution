namespace = lpr_events

# ============================================================================
# POLITICS LAW ENACTMENT EVENTS
# ============================================================================
# These events can be triggered during law enactment debates, allowing players
# to spend LPR points for political advantages. Each event offers different
# benefits but comes with costs and drawbacks.
# ============================================================================

# Politics event for Liberty - triggered during law enactment debates
lpr_events.30 = {
    type = country_event
    placement = root

    category = enactment

    trigger = {
        # Law enactment check removed - not a standard Victoria 3 trigger
        has_variable = actionables_lpr_liberty
        var:actionables_lpr_liberty >= 5
    }

    title = lpr_events.30.t
    desc = lpr_events.30.d
    flavor = lpr_events.30.f

    event_image = {
        video = "unspecific_politicians_arguing"
    }

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
    icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

    duration = 7

    # Option A: Spend Liberty for Political Advantage
    # Benefits: Increases enactment advance and debate chances
    # Costs: Liberty points, gold, radicalizes Trade Unions
    option = {
        name = lpr_events.30.a
        lpr_politics_liberty_effect = yes
    }

    # Option B: Decline the Political Maneuver
    # No benefits or costs
    option = {
        name = lpr_events.30.b
        default_option = yes
    }
}

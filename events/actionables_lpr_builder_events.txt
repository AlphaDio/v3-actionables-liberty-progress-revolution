namespace = lpr_events

# ============================================================================
# BUILDER EVENTS (Monthly Events)
# ============================================================================
# These are the core monthly events that allow players to accumulate LPR points.
# Each event fires every month (100% chance) and offers three choices:
# - Liberty: Focuses on independence and freedom
# - Progress: Focuses on technological and social advancement
# - Revolution: Focuses on radical change and upheaval
#
# Each choice grants:
# - 3 points in the corresponding local LPR variable
# - 1 point in the corresponding global variable
# - Additional immediate rewards (money, modifiers, etc.)
#
# After accumulating 12 points in any category, a finisher event triggers.
# ============================================================================

# Monthly builder event - fires every month for all countries
lpr_events.1 = {
    type = country_event
    placement = root

    title = lpr_events.1.t
    desc = lpr_events.1.d
    flavor = lpr_events.1.f

    event_image = {
        video = "unspecific_world_fair"
    }

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
    icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

    duration = 3

    # Option A: Liberty Path
    # Rewards: 3 Liberty points (local) + 1 Independence point (global)
    # Drawbacks: Small treasury cost + 3-month radical increase (economic freedom causes social unrest)
    # Focus: Economic freedom and national independence
    option = {
        name = lpr_events.1.a
        add_treasury = -2000
        add_modifier = {
            name = lpr_small_radical_increase
            months = 3
        }
        change_variable = {
            name = actionables_lpr_liberty
            add = 3
        }
        change_global_variable = {
            name = actionables_lpr_independence
            add = 1
        }
    }

    # Option B: Progress Path
    # Rewards: 30-day construction speed boost (3x multiplier) + 3 Progress points (local) + 1 Innovation point (global)
    # Drawbacks: Larger treasury cost for infrastructure investment + 3-month radical increase from rapid change
    # Focus: Technological advancement and innovation through rapid construction
    option = {
        name = lpr_events.1.b
        add_treasury = -8000
        add_modifier = {
            name = lpr_very_small_radical_increase
            months = 3
        }
        add_modifier = {
            name = lpr_construction_boost
            days = 30
        }
        change_variable = {
            name = actionables_lpr_progress
            add = 3
        }
        change_global_variable = {
            name = actionables_lpr_innovation
            add = 1
        }
    }

    # Option C: Revolution Path
    # Rewards: 30-day society tech spread boost modifier + 3 Revolution points (local) + 1 Uprising point (global)
    # Drawbacks: Treasury cost + 3-month radical increase (revolutionary ideas cause major unrest) + population loss from uprising
    # Focus: Social change and revolutionary ideas
    option = {
        name = lpr_events.1.c
        add_treasury = -3000
        add_modifier = {
            name = lpr_medium_radical_increase
            months = 3
        }
        random_scope_state = {
            limit = {
                any_scope_pop = {
                    size > 15000
                }
            }
            random_scope_pop = {
                limit = {
                    size > 15000
                }
                change_pop_size = -5000
            }
        }
        add_modifier = {
            name = lpr_society_tech_spread_boost
            days = 30
        }
        change_variable = {
            name = actionables_lpr_revolution
            add = 3
        }
        change_global_variable = {
            name = actionables_lpr_uprising
            add = 1
        }
    }
}

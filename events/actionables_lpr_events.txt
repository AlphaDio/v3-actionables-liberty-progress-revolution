namespace = actionables_lpr_event

# Initialization Event
actionables_lpr_event.100 = {
    type = country_event
    placement = root

    title = actionables_lpr_event.100.t
    desc = actionables_lpr_event.100.d

    event_image = {
        video = "europenorthamerica_springtime_of_nations"
    }

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
    icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

    trigger = {
        NOT = { has_global_variable = lpr_initialized }
    }

    immediate = {
        set_variable = { name = liberty_points value = 0 }
        set_variable = { name = progress_points value = 0 }
        set_variable = { name = revolution_points value = 0 }
        set_global_variable = { name = lpr_initialized value = 1 }
    }

    option = {
        name = actionables_lpr_event.100.a
    }
}

actionables_lpr_event.101 = {
    type = country_event
    placement = root

    title = actionables_lpr_event.101.t
    desc = actionables_lpr_event.101.d

    event_image = {
        video = "unspecific_military_parade"
    }

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

    icon = "gfx/interface/icons/event_icons/event_military.dds"

    duration = 1
    cooldown = { days = 1 }

    trigger = {
        always = yes
    }

    # LIBERTY option
    option = {
        name = actionables_lpr_event.101.a.n
        change_variable = { 
            name = liberty_points 
            add = 10
        }
        add_treasury = 1000
        ai_chance = {
            base = 25
        }
    }

    # PROGRESS option
    option = {
        name = actionables_lpr_event.101.b.n
        change_variable = { 
            name = progress_points 
            add = 10
        }
        add_radicals = { value = 10 }
        ai_chance = {
            base = 25
        }
    }

    # REVOLUTION option
    option = {
        name = actionables_lpr_event.101.c.n
        change_variable = { 
            name = revolution_points 
            add = 10
        }
        change_infamy = 10
        ai_chance = {
            base = 25
        }
    }

    # Default option
    option = {
        name = actionables_lpr_event.101.d.n
        default_option = yes
        change_variable = { 
            name = liberty_points 
            add = 1 
        }
        change_variable = { 
            name = progress_points 
            add = 1 
        }
        change_variable = { 
            name = revolution_points 
            add = 1 
        }
        ai_chance = {
            base = 25
        }
    }
}

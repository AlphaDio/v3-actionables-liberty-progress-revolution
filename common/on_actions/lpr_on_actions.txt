on_monthly_pulse_country = {
	on_actions = {
		lpr_general_builder_events
		lpr_check_finisher_monthly
	}
}

# General builder events - 100% chance per month (guaranteed for testing)
lpr_general_builder_events = {
	random_events = {
		chance_to_happen = 100
		1 = lpr_events.1
	}
}

# Check for finisher events every month (triggers if conditions met)
lpr_check_finisher_monthly = {
	effect = {
		lpr_check_finisher_events = yes
	}
}

# General finisher events - triggered if LPR variables meet conditions
lpr_general_finisher_events = {
	random_events = {
		chance_to_happen = 100
		1 = lpr_events.finisher.1
	}
}

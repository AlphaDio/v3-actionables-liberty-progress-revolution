# ============================================================================
# GLOBAL HISTORY INITIALIZATION
# ============================================================================
# This file initializes all LPR (Liberty, Progress, Revolution) variables
# at game start. It runs last among all history files to ensure proper setup.
#
# Variable System:
# - Global Variables (IIT): Track worldwide trends, shared across all countries
#   * actionables_lpr_independence: Global measure of independence movements
#   * actionables_lpr_innovation: Global measure of technological progress
#   * actionables_lpr_uprising: Global measure of revolutionary sentiment
#
# - Local Variables (LPR): Track individual country progress
#   * actionables_lpr_liberty: Country's liberty/independence focus
#   * actionables_lpr_progress: Country's technological progress focus
#   * actionables_lpr_revolution: Country's revolutionary focus
#
# These variables are modified by monthly builder events and consumed by
# finisher events when thresholds (12 points) are reached.
# ============================================================================

# This is executed last among all history files
GLOBAL = {
	# Initialize global variables (IIU - Independence, Innovation, Uprising)
	# These track worldwide trends and are shared across all countries
	set_global_variable = {
		name = actionables_lpr_independence
		value = 0
	}
	set_global_variable = {
		name = actionables_lpr_innovation
		value = 0
	}
	set_global_variable = {
		name = actionables_lpr_uprising
		value = 0
	}
	
	# Initialize local LPR variables for each country
	# These track individual country progress toward different paths
	every_country = {
		set_variable = {
			name = actionables_lpr_liberty
			value = 0
		}
		set_variable = {
			name = actionables_lpr_progress
			value = 0
		}
		set_variable = {
			name = actionables_lpr_revolution
			value = 0
		}
		
		# Initialize cumulative LPR spent variables
		# These track total amount of each LPR type spent over the entire game
		set_variable = {
			name = actionables_lpr_liberty_spent
			value = 0
		}
		set_variable = {
			name = actionables_lpr_progress_spent
			value = 0
		}
		set_variable = {
			name = actionables_lpr_revolution_spent
			value = 0
		}
	}
	
	# Add journal entry for player countries to track LPR points
	# This provides an in-game UI element showing current point values
	every_country = {
		limit = {
			is_player = yes
		}
		add_journal_entry = {
			type = je_lpr_points
		}
	}
}

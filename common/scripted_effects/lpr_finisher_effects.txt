# Finisher event system - triggers every 3 months if LPR variables meet conditions
lpr_check_finisher_events = {
	if = {
		limit = {
			OR = {
				var:actionables_lpr_liberty >= 2
				var:actionables_lpr_progress >= 2
				var:actionables_lpr_revolution >= 2
			}
		}
		lpr_general_finisher_events = yes
	}
}
